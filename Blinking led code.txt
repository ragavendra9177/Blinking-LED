#include <stdint.h>
#include <stdlib.h>



/* ================= Register Addresses ================= */

// RCC AHB1 Peripheral Clock Enable Register
#define RCC_AHB1ENR   (*(volatile uint32_t *)0x40023830)

// GPIOA Registers
#define GPIOA_MODER   (*(volatile uint32_t *)0x40020000)
#define GPIOA_IDR     (*(volatile uint32_t *)0x40020010)
#define GPIOA_ODR     (*(volatile uint32_t *)0x40020014)


/* ================= Main Function ================= */

int main(void)
{
    /* -------- 1. Enable GPIOA Clock -------- */
    RCC_AHB1ENR |= (1 << 0);   // Enable clock for GPIOA


    /* -------- 2. Configure PA9 as Output -------- */
    // Each pin uses 2 bits in MODER
    // PA9 → bits [19:18]

    GPIOA_MODER &= ~(3 << 18); // Clear bits (make 00)
    GPIOA_MODER |=  (1 << 18); // Set as 01 → Output mode


    /* -------- 3. Configure PA0 as Input -------- */
    // PA0 → bits [1:0]
    // 00 → Input mode (default, but we set explicitly)

    GPIOA_MODER &= ~(3 << 0);  // Clear bits → Input mode


    /* -------- 4. Infinite Loop -------- */
    while(1)
    {
        /* Read Button State (PA0) */

        // If PA0 is LOW → Button pressed (Active LOW)
        if( (GPIOA_IDR & (1 << 0)) == 0 )
        {
            // LED ON (Active LOW → write 0)
            GPIOA_ODR &= ~(1 << 9);
        }
        else
        {
            // LED OFF (write 1)
            GPIOA_ODR |= (1 << 9);
        }
    }
}
